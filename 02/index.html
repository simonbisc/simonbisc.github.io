y = data["GOOG"].pct_change()

y.dropna(inplace=True)
coeff = b, a = np.polyfit(x, y, 1)

# Rsquared  reg dev / total dev
p = np.poly1d(coeff)
yi = p(x)
ym = np.sum(y)/len(y)
devreg = np.sum((yi-ym)**2)
devtot = np.sum((y - ym)**2)
Rsquared = devreg / devtot
